	fun ComparaLunghezzaListe<X> : (list[X],list[X])->list[X] (listaA, listaB){
		fun lunghezzaLista<X> : (list[X])->int (lista){
			if (lista == empty) then{0}
			else { lunghezzaLista<X>(rest(lista)) }
		};

		then {lunghezzaLista<X>(listaA)}
		else { 0 }	
	};
