let fun prova:int (l:int, check:bool)
	{
		var x:int = (l/2)*4;
		var y:int= l*2;
		fun arrow:(int)->int (p)
		{
			y
		};
		
		fun nested:int (x:int)
		{
			var y:int = l+(x-(y/2));
			x-(y/2)
		};
		if ( (check==true) && ( nested(x) <= l-1 ) )
		then {	arrow(1)-x	}
		else {	x+y	}
	};
in print(prova(10,true));
